  • 1. Tab Index 

    // set tabindex -1 attribute for anchor tag with # value || button url is same as current page url
jQuery('a').each(function() {
    // console.log(Window.location.href);
    if (this.getAttribute('href') === '#' || this.href === window.location.href) {
    jQuery(this).attr('tabindex', '-1');
    }
});




// for disable focus set "tabindex= -1" for LI tag
$(".fasd_header .main_header li").each(function () {
    $(this).attr("tabindex", "-1");
});
// for enable focus set  "tabindex= 0" for a tag
$(".fasd_header .main_header li").each(function () {
    $(this).find("a").attr("tabindex", "0");
});

  • 3. Change Owl carousel arrows button attribute 

    // Change role ="presentation" attribute to "button"
jQuery(".owl-prev").each(function (index) {
    jQuery(this).attr("aria-label", "Previous Slide");
    jQuery(this).attr("role", "button");
});
jQuery(".owl-next").each(function (index) {
    jQuery(this).attr("aria-label", "Next Slide");
    jQuery(this).attr("role", "button");
});

  • 4. Owl carousel slides change with keyboard's arrow button 

    //Owl carousel slides change with keyboard's arrow button 
jQuery(document.documentElement).keyup(function (event) {
    // Select all elements with the class .owl-carousel
    jQuery(".owl-carousel").each(function () {
    var owl = jQuery(this);
    console.log(owl);
    owl.owlCarousel();

    // handle cursor keys
    if (event.keyCode == 37) {
        // go left
        console.log("left");
        owl.trigger("prev.owl.carousel", [300]);
    } else if (event.keyCode == 39) {
        // go right
        console.log("right");
        owl.trigger("next.owl.carousel");
    }
    });
});
