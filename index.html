function setVideoDuration() {
    let $videoSrc = $(".video_duration_src");
    let $videoDuration = $(".video_duration");

    // Wait for the video metadata to load
    $videoSrc.on("loadedmetadata", function () {
    // Get the duration of the video in seconds
    const duration = Math.floor(this.duration);

    // Format the duration into HH:MM:SS format
    const formattedDuration = formatVideoDuration(duration);

    // Set the formatted duration to all .video_duration elements
    $videoDuration.each(function () {
        $(this).html(formattedDuration);
    });

    // console.log(formattedDuration);
    });
}

function nextStepFun(e) {
    promotionFormValidateFun();

        promotionStepCount = promotionStepCount + 1;
        promotionStepFormActive();
    }

function promotionFormValidateFun(e) {
    let allRequiredFields, anyError;
    allRequiredFields = promotion_step_form_steps[promotionStepCount].querySelectorAll(".required");
    allRequiredFields.forEach((currField) => {
        promotionFieldValidateFun(currField);
    });
    anyError = promotion_step_form_steps[promotionStepCount].querySelector(".error");
    if (anyError) {
        isPromotionsFormValidate = false;
    } else {
        isPromotionsFormValidate = true;
    }
}
function promotionFieldValidateFun(currField) {
    let currFieldVal = currField.value;
    if (currFieldVal == "") {
        promotionFieldSetError(currField, `${errorsMessage.reqMsg}`);
    } else {
        promotionFieldSetSuccess(currField);
    }
    if (currField.classList.contains("email")) {
        var mailformat = /^\w+([\.-]?\w+)*@\w+([\.-]?\w+)*(\.\w{2,3})+$/;
        if (currFieldVal == "") {
            promotionFieldSetError(currField, `${errorsMessage.emailEnterMsg}`);
        } else if (!currFieldVal.match(mailformat)) {
            promotionFieldSetError(currField, `${errorsMessage.validEmailMsg}`);
        } else {
            promotionFieldSetSuccess(currField);
        }
    }
}
function promotionFieldSetError(currField, msg) {
    let currFieldParent, errorPresent, errorSpan;
    currFieldParent = currField.parentElement;
    errorPresent = currFieldParent.querySelector(".error");
    if (!errorPresent) {
        errorSpan = document.createElement("span");
        errorSpan.classList.add("error");
        errorSpan.textContent = msg;
        currFieldParent.appendChild(errorSpan);
    } else {
        errorPresent.textContent = msg;
    }
}
function promotionFieldSetSuccess(currField) {
    let currFieldParent, errorPresent;
    currFieldParent = currField.parentElement;
    errorPresent = currFieldParent.querySelector(".error");
    if (errorPresent) {
        errorPresent.remove();
    }
}
function promotionStepFormActive(e) {
    let stepsLength = promotion_step_form_steps.length;
    if (promotionStepCount < stepsLength) {
        promotion_step_form_steps.forEach((step) => {
            step.classList.remove("active");
        });
        promotion_step_form_steps[promotionStepCount].classList.add("active");
    }
}

            is_any_another_form_in_future();
        }

        console.log("Error while submit");
    }

  • 13. Set UTM Parameters 

    // Set UTM parameters
// Function to get URL parameters
function getUrlParams() {
  debugger;
  const params = {};
  const queryString = window.location.search;
  const urlParams = new URLSearchParams(queryString);
  urlParams.forEach((value, key) => {
    params[key] = value;
  });
  return params;
}

// Function to set hidden form fields
function setHiddenFields() {
  const params = getUrlParams();

  const utmSourceField = document.querySelector('input[name="utm_source"]');
  const utmMediumField = document.querySelector('input[name="utm_medium"]');
  const utmCampaignField = document.querySelector('input[name="utm_campaign"]');
  const utmTermField = document.querySelector('input[name="utm_term"]');
  const utmContentField = document.querySelector('input[name="utm_content"]');

  if (utmSourceField && params.utm_source) {
    utmSourceField.value = params.utm_source;
  }
  if (utmMediumField && params.utm_medium) {
    utmMediumField.value = params.utm_medium;
  }
  if (utmCampaignField && params.utm_campaign) {
    utmCampaignField.value = params.utm_campaign;
  }
  if (utmTermField && params.utm_term) {
    utmTermField.value = params.utm_term;
  }
  if (utmContentField && params.utm_content) {
    utmContentField.value = params.utm_content;
  }
}

// Call the function to set hidden fields on page load
document.addEventListener("DOMContentLoaded", setHiddenFields);
