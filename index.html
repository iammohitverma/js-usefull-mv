  • 4. Owl carousel slides change with keyboard's arrow button 

    //Owl carousel slides change with keyboard's arrow button 
jQuery(document.documentElement).keyup(function (event) {
    // Select all elements with the class .owl-carousel
    jQuery(".owl-carousel").each(function () {
    var owl = jQuery(this);
    console.log(owl);
    owl.owlCarousel();

    // handle cursor keys
    if (event.keyCode == 37) {
        // go left
        console.log("left");
        owl.trigger("prev.owl.carousel", [300]);
    } else if (event.keyCode == 39) {
        // go right
        console.log("right");
        owl.trigger("next.owl.carousel");
    }
    });
});

      // Function to handle setting video duration
function setVideoDuration() {
    let $videoSrc = $(".video_duration_src");
    let $videoDuration = $(".video_duration");

    // Wait for the video metadata to load
    $videoSrc.on("loadedmetadata", function () {
    // Get the duration of the video in seconds
    const duration = Math.floor(this.duration);

    // Format the duration into HH:MM:SS format
    const formattedDuration = formatVideoDuration(duration);

    // Set the formatted duration to all .video_duration elements
    $videoDuration.each(function () {
        $(this).html(formattedDuration);
    });

    // console.log(formattedDuration);
    });
}

// Call the function to set video duration initially
setVideoDuration();

// Rebind the event handler after AJAX call
$(document).ajaxComplete(function () {
  setVideoDuration();
});

  • 8. JustCo step form validation 

     let promotion_step_form = document.querySelector(".step_form_container form");
let promotion_step_form_steps = document.querySelectorAll(".step_form_container .step");
let promotion_step_form_next_btns = document.querySelectorAll(".step_form_container .next_step");
let promotionStepCount = 0;
let isPromotionsFormValidate;
promotion_step_form_next_btns.forEach((nextBtn) => {
    nextBtn.addEventListener("click", nextStepFun);
});
function nextStepFun(e) {
    promotionFormValidateFun();
    if (isPromotionsFormValidate == true) {
        promotionStepCount = promotionStepCount + 1;
        promotionStepFormActive();
    }
}
function promotionFormValidateFun(e) {
    let allRequiredFields, anyError;
    allRequiredFields = promotion_step_form_steps[promotionStepCount].querySelectorAll(".required");
    allRequiredFields.forEach((currField) => {
        promotionFieldValidateFun(currField);
    });
    anyError = promotion_step_form_steps[promotionStepCount].querySelector(".error");
    if (anyError) {
        isPromotionsFormValidate = false;
    } else {
        isPromotionsFormValidate = true;
    }
}
function promotionFieldValidateFun(currField) {
    let currFieldVal = currField.value;
    if (currFieldVal == "") {
        promotionFieldSetError(currField, `${errorsMessage.reqMsg}`);
    } else {
        promotionFieldSetSuccess(currField);
    }
    if (currField.classList.contains("email")) {
        var mailformat = /^\w+([\.-]?\w+)*@\w+([\.-]?\w+)*(\.\w{2,3})+$/;
        if (currFieldVal == "") {
            promotionFieldSetError(currField, `${errorsMessage.emailEnterMsg}`);
        } else if (!currFieldVal.match(mailformat)) {
            promotionFieldSetError(currField, `${errorsMessage.validEmailMsg}`);
        } else {
            promotionFieldSetSuccess(currField);
        }
    }
}
function promotionFieldSetError(currField, msg) {
    let currFieldParent, errorPresent, errorSpan;
    currFieldParent = currField.parentElement;
    errorPresent = currFieldParent.querySelector(".error");
    if (!errorPresent) {
        errorSpan = document.createElement("span");
        errorSpan.classList.add("error");
        errorSpan.textContent = msg;
        currFieldParent.appendChild(errorSpan);
    } else {
        errorPresent.textContent = msg;
    }
}
function promotionFieldSetSuccess(currField) {
    let currFieldParent, errorPresent;
    currFieldParent = currField.parentElement;
    errorPresent = currFieldParent.querySelector(".error");
    if (errorPresent) {
        errorPresent.remove();
    }
}
function promotionStepFormActive(e) {
    let stepsLength = promotion_step_form_steps.length;
    if (promotionStepCount < stepsLength) {
        promotion_step_form_steps.forEach((step) => {
            step.classList.remove("active");
        });
        promotion_step_form_steps[promotionStepCount].classList.add("active");
    }
}
if (promotion_step_form) {
    promotion_step_form.addEventListener("submit", promotionFormSubmitFun);
}
function promotionFormSubmitFun(event) {
    promotionFormValidateFun();
    event.preventDefault();
    let currForm = event.target;
    let formId = currForm.id;
    let errorPresent = currForm.querySelector(".error");
    if (!errorPresent) {
        promotionStepCount = 0;
        if (formId == "business_referral_form") {
            is_business_referral_form_submit();
        }
        if (formId == "any_another_form_in_future") {
            is_any_another_form_in_future();
        }
    } else {
        console.log("Error while submit");
    }
}
function is_business_referral_form_submit() {
    console.log("form is submitted");
}

$('[name="state"]').change(function () {
    let allValues = $(this).val();
    console.log(allValues);
    let allChoices = $(this)
    .closest(".input_wrap")
    .find(".chosen-container ul li.search-choice");
    $("#stateAccordion .chosen-choices").append(allChoices);

    if (allValues.length > 0) {
        $("#stateAccordion .accordion-button").removeAttr("disabled");
        $("#stateAccordion .accordion-button").removeClass("collapsed");
        $("#stateAccordion .accordion-collapse").addClass("show");
        $('[name="state"] + div input').attr("readonly", "readonly");
        $('[name="state"] + div .selectedPlaceholder').show();

        $("#stateAccordion .accordion-button").attr("disabled", "disabled");

        $("#stateAccordion .accordion-collapse").removeClass("show");
        $('[name="state"] + div input').removeAttr("readonly");

    }
});
